\def\stylerefnetwork#1{%
	\setupbodyfont[karlaregular, 5pt]
	\setupinterlinespace[7pt] 
	\setupalign[flushleft]
	\setcharacterspacing[frenchpunctuation]
	\defineblank[default][0mm]
	\setuptolerance[verytolerant]s 
	\color[black]
	#1
}
