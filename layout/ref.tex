{
	\margintext{\kasml #1}
}

\def\stylerefroadback#1%
{
	{\inright{\kasml #1}}
}

\def\stylerefroadforward#1%
{
	{\inright{\kasml #1}}
}

\def\stylerefroad#1#2%
{
	\margintext{
		\hskip -1.5mm
		{\kasml
			\vbox to 8pt {
				\hbox{#1}
				\vskip -5pt
				\hbox{#2}
			}
		}
	}
}

\def\styleref#1#2#3%
{
	if #1 == \getvariable{type}{mail} then
		if #2 == \getvariable{road}{slow} then
			{\crimslow #3}
		elseif #2 == \getvariable{road}{cash} then
			{\crimcash #3}
		elseif #2 == \getvariable{road}{patch} then
			{\crimpatch #3}
		elseif #2 == \getvariable{road}{chat} then
			{\crimchat #3}
		elseif #2 == \getvariable{road}{red} then
			{\crimred #3}
		elseif #2 == \getvariable{road}{science} then
			{\crimscience #3}
		elseif #2 == \getvariable{road}{themes} then
			{\crimthemes #3}
		elseif #2 == \getvariable{road}{network} then
			{\crimnetwork #3}
	else
		% A kind of inconsistent specification, but used
		% in for instance s-pre-19.tex; the cutspace is
		% used only for determining some kind of right
		% margin; don't use this in doublesided mode
		pass 
	end
}
